### 素数筛法

#### 欧拉筛

```cpp
vector<int> get_primes(int n) {
    vector<int> p;
    vector<bool> vis(n + 1);
    for (int i = 2; i <= n; ++i) {
        if (!vis[i]) p.push_back(i);
        for (int j = 0; j < p.size() && p[j] <= n / i; ++j) {// i * p[j] <= n 容易溢出
            vis[i * p[j]] = true;
            if (i % p[j] == 0) break;
        }
    }
    return p;
}
```

###### 证明

```
先证正确性：对合数 x = py，其中 p 是 x 的最小质因子。
考虑当 i = y 时，枚举已筛出的质数 j。由于 p 是最小质因
子，所以 i 不含小于 p 的质因子，所以在枚举到 p 前，
i mod j = 0 不会成立，循环不会终止。这就证明了 x 会被筛掉。
再证明复杂度为线性：假设 x = py = qz，其中 p 是 x 的最
小质因子，q 是另一质因子。
根据唯一分解定理，z 一定含有质因子 p，且 p < q。则
i = z, j = p 时，i mod j = 0 会先一步成立，循环终止，质数不会
枚举到 q。
考察证明过程可以发现，j 就是合数 i × j 的 最小质因子。
```

##### 欧拉筛求最小质因子

```cpp
vector<int> get_min_primes(int n) {
    vector<int> p;
    vector<int> mp(n + 1);//最小质因子
    vector<bool> vis(n + 1);
    for (int i = 2; i <= n; ++i) {
        if (!vis[i]) {
            p.push_back(i);
            mp[i] = i;// 如果是质数最小质因子是本身
        }
        for (int j = 0; j < p.size() && p[j] <= n / i; ++j) {
            vis[i * p[j]] = true;
            mp[i * p[j]] = p[j];// 如果是合数则最小质因子恰好是p[j]
            if (i % p[j] == 0) break;
        }
    }
    return mp;
}
```

##### 欧拉筛求欧拉函数

```cpp
vector<int> get_phi(int n) {
    vector<int> p;
    vector<bool> vis(n + 1);
    vector<int> phi(n + 1);
    phi[1] = 1;
    for (int i = 2; i <= n; ++i) {
        if (!vis[i]) {
            p.push_back(i);
            phi[i] = i - 1;
        }
        for (int j = 0; j < p.size() && p[j] <= n / i; ++j) {
            vis[i * p[j]] = true;
            if (i % p[j] == 0) {
                phi[i * p[j]] = phi[i] * p[j];
                // p[j] 是 i*p[j],i的最小质因子
                break;
            } else {
                phi[i * p[j]] = phi[i] * (p[j] - 1);
                // gcd(i,p[j]) == 1 积性函数性质 phi[p[j]] = p[j] - 1
            }
        }
    }
    return phi;
}
```
