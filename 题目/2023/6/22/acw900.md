#include <bits/stdc++.h>
const int mod = 1e9 + 7;
const int N = 6;
int dp[N][N];

int main() {
    int n;
    std::cin >> n;

    for (int i = 0; i <= n; ++i) dp[i][0] = 1;
    for (int j = 1; j <= n; ++j) dp[0][j] = 0;

    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= n; ++j) {

            if (j - i >= 0)
                dp[i][j] = (dp[i - 1][j] + dp[i][j - i]) % mod;
            else
                dp[i][j] = dp[i - 1][j];
        }


    std::cout << dp[n][n] << "\n";

//    for (int i = 0; i <= n; ++i)
//        for (int j = 0; j <= n; ++j)
//            std::cout << dp[i][j] << " \n"[j == n];

    return 0;
}
