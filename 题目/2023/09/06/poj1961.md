> kmp 

```cpp
#include <bits/stdc++.h>
using namespace std;

const int N = 1e6 + 10;
char s[N];
int p[N];
int n;
void getPi() {
    for (int i = 2, j = 0; i <= n; ++i) {
        while (j != 0 && s[j + 1] != s[i]) j = p[j];
        if (s[j + 1] == s[i]) p[i] = ++j;
    }
}

int main() {
    int kase = 0;
    while (scanf("%d", &n) != EOF && n != 0) {
        memset(s, 0, sizeof s);
        memset(p, 0, sizeof p);
        scanf("%s", s + 1);
        getPi();
        printf("Test case #%d\n", ++kase);
        for (int i = 2; i <= n; ++i) {

            int pl = i - p[i]; // period length
            if (i % pl == 0 && p[i] != 0) {
                printf("%d %d\n", i, i / pl);

            }

        }
        printf("\n");
    }
    return 0;
}
```
