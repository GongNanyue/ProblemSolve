有向无环图dp



```cpp




#include <bits/stdc++.h>
using namespace std;
using ll = long long;

#ifndef ONLINE_JUDGE
const int N = 5;
#else
const int N = 1e3+10;
#endif

ll a[N][N];
ll vl[N][N];


int n, m;

ll calPort(int x, int y) {
    ll f[x + 1][y + 1];
    f[x][y] = a[x][y];
    for (int i = x; i >= 1; --i)
        for (int j = y; j >= 1; --j) {
            if (i + 1 <= x && j + 1 <= y)
                f[i][j] = max(f[i + 1][j], f[i][j + 1]) + a[i][j];
            else if (i + 1 <= x)
                f[i][j] = f[i + 1][j] + a[i][j];
            else if (j + 1 <= y)
                f[i][j] = f[i][j + 1] + a[i][j];
            else
                f[i][j] = a[i][j];
        }
    return f[1][1];
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j)
            cin >> a[i][j];
    vl[n][m] = a[n][m];

    for (int i = n; i >= 1; --i)
        for (int j = m; j >= 1; --j) {
            if (i + 1 <= n && j + 1 <= m)
                vl[i][j] = max(vl[i + 1][j], vl[i][j + 1]) + a[i][j];
            else if (i + 1 <= n)
                vl[i][j] = vl[i + 1][j] + a[i][j];
            else if (j + 1 <= m)
                vl[i][j] = vl[i][j + 1] + a[i][j];
            else
                vl[i][j] = a[i][j];
        }


    int t;
    cin >> t;
    while (t--) {
        vector<pair<int, int>> port;
        map<pair<int, int>, ll> pval;
        ll res = vl[1][1];
        int k;
        cin >> k;
        for (int i = 0; i < k; ++i) {
            int x, y;
            cin >> x >> y;
            port.push_back({x, y});
            pval[{x, y}] = calPort(x, y);
        }
        sort(port.begin(), port.end());
    
        for (int i = 0; i < port.size(); ++i)
            for (int j = 0; j < port.size(); ++j) {
                if(i != j){
                    int x1 = port[i].first, y1 = port[i].second, x2 = port[j].first, y2 = port[j].second;
                    res = max(res, pval[{x2, y2}] + vl[x1][y1]);
                }


            }
        cout << res << "\n";
    }


    return 0;
}
```
