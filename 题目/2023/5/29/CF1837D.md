```cpp
#include <bits/stdc++.h>
#define yes cout<<"Yes"<<endl
#define no cout<<"No"<<endl

using namespace std;
typedef long long ll;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        unordered_map<char,int> um;
        int cnt1=0,cnt2=0;
        um['(']=1,um[')']=-1;
        vector<int> a(n+1);
        for(int i=1;i<=n;i++){
            char c;
            cin>>c;
            a[i]=um[c];
            if(a[i]==1) cnt1++;
            if(a[i]==-1)cnt2++;
        }
        if(cnt1!=cnt2){
            cout<<-1<<endl;
        }
        else{
            vector<int> s(n+1,0);
            for(int i=1;i<=n;i++){
                s[i]=s[i-1]+a[i];
            }
            vector<int> ans(n+1,1);
            bool flag1=false,flag2=false;
            for(int i=1;i<=n;i++){
                if(s[i]>0) {
                    ans[i]=1;
                    flag1= true;
                }
                if(s[i]<0) {
                    ans[i]=2;
                    flag2= true;
                }
                if(s[i]==0)ans[i]=ans[i-1];
            }
            if(flag1+flag2==1){
                cout<<1<<endl;
                for(int i=1;i<=n;i++){
                    cout<<1<<" \n"[i==n];
                }
            }
            else{
                cout<<flag1+flag2<<endl;
                for(int i=1;i<=n;i++){
                    cout<<ans[i]<<" \n"[i==n];
                }
            }
        }

    }


    return 0;
}
```

官方形式做法
```cpp
#include <bits/stdc++.h>
using namespace std;


struct interval {
    //[l,r] state 1->reverse
    int l, r, state;

    interval(int a = 0, int b = 0, int c = 0) : l(a), r(b), state(c) {}

    bool operator==(const interval &other) const {
        return l == other.l && r == other.r && state == other.state;
    }


    bool operator<(const interval &other) const {
        return tie(l, r, state) < tie(other.l, other.r, other.state);
    }
};

int trans(char c) {
    return c == '(' ? 1 : -1;
}


int main() {
    int t;
    cin >> t;
    while (t--) {
        vector<interval> iv, nv;
        int n;
        cin >> n;
        string s;
        cin >> s;
        int i, j, cnt;
        for (i = 0, j = 0; i < n && j < n;) {
            j = i + 1;
            cnt = trans(s[i]);
            while (j < n && cnt != 0) {
                ++j;
                cnt += trans(s[j - 1]);
            }
            iv.emplace_back(i, j - 1, ((cnt - trans(s[j - 1])) > 0) ? 0 : 1);
            i = j;
        }
        for (i = 0, j = 0; i < iv.size() && j < iv.size();) {
            j = i + 1;
            while (j < iv.size() && iv[j].state == iv[i].state) ++j;
            nv.emplace_back(iv[i].l, iv[j - 1].r, iv[i].state);
            i = j;
        }
        for (i = 0, cnt = 0; i < n; ++i)
            cnt += trans(s[i]);

        if (cnt != 0)
            cout << -1 << "\n";
        else {
            if (nv.size() == 1) {
                cout << 1 << "\n";
                for (int k = 0; k < n; ++k)
                    cout << 1 << (k == n - 1 ? "\n" : " ");
            } else {
                cout << 2 << "\n";
                for (const auto &interVal: nv)
                    for (int k = interVal.l; k <= n && k <= interVal.r; ++k) {
                        cout << (interVal.state == 0 ? 1 : 2) << " ";
                    }
                cout << "\n";
            }
        }

    }

    return 0;
}
```
